set自定义比较函数
发布于：2023-10-19 23:29:07 标签：/ sylar / 访问：

使用insert()将元素插入到集合中去的时候，集合会根据设定的比较函数将该元素放到该放的节点上去。在定义集合的时候，如果没有指定比较函数，那么采用默认的比较函数(按键值由小到大的顺序插入元素)。

自定义数据结构时，在使用 std::set 的过程中， std::set 会默认调用自定义数据结构的 operator< 操作符函数来比较两个对象的大小来进行排序（std::set 是有序的关联容器）

重载<操作符

记住一个准则：永远让比较函数对相同元素返回false。
set容器在判定已有元素a和新插入元素b是否相等时，是这么做的：1）将a作为左操作数，b作为有操作数，调用比较函数，并返回比较值 2）将b作为左操作数，a作为有操作数，再调用一次比较函数，并返回比较值。如果1、2两步的返回值都是false，则认为a、b是相等的，则b不会被插入set容器中；如果1、2两步的返回值都是true，则可能发生未知行为，因此，记住一个准则：永远让比较函数对相同元素返回false。

我们的操作符函数需要被定义成安全的 const 成员函数，其中 std::set 默认调用了不修改自定结构的值的 operator< 操作符，因此我们应该提供 const 版本的操作符重载版本：


// 按照id去重，按照score排序
#include <iostream>
#include <set>
using namespace std;
struct Person {
  int m_id;
  int m_score;
  Person(int id, int score) {
    this->m_id = id;
    this->m_score = score;
  }
//   我们的操作符函数需要被定义成安全的 const 成员函数，其中 std::set 默认调用了不修改自定结构的值的 operator< 操作符，因此我们应该提供 const 版本的操作符重载版本：
  bool operator<(const struct Person& right) const  //重载<运算符
  {
    if (this->m_id == right.m_id)  //根据id去重
      return false;
    else {
      if (this->m_score != right.m_score) {
        return this->m_score > right.m_score;  //按照分数降序，排序
      } else {
        return this->m_id > right.m_id;
      }
    }
  }
};

int main() {
  std::set<Person> mySet;
  Person s1(10, 100);
  Person s2(20, 200);
  Person s3(20, 300);
  Person s4(30, 200);
  Person s5(30, 300);
  mySet.insert(s1);  //插入s1
  mySet.insert(s2);  //插入s2
  mySet.insert(s3);  // s3和s2的id相同，不插入
  mySet.insert(s4);  //插入s4
  for (auto it : mySet) {
    std::cout << "id:" << it.m_id << ",score:" << it.m_score << std::endl;
  }
  std::cout << "end" << std::endl;
  return 0;
};
结果如下：
id:30,score:200
id:20,score:200
id:10,score:100

#include <iostream>
#include <set>
using namespace std;
struct Person {
  int m_id;
  int m_score;
  Person(int id, int score) {
    this->m_id = id;
    this->m_score = score;
  }

  // bool operator<(const struct Person& right) const  //重载<运算符
  // {
  //   if (this->m_id == right.m_id)  //根据id去重
  //     return false;
  //   else {
  //     if (this->m_score != right.m_score) {
  //       return this->m_score > right.m_score;  //降序
  //     } else {
  //       return this->m_id > right.m_id;
  //     }
  //   }
  // }
};

struct Cmp {
  bool operator()(const Person &lhs, const Person &rhs) const {
    if (lhs.m_id == rhs.m_id) {
      return false;
    } else {
      if (lhs.m_score != rhs.m_score) {
        return lhs.m_score > rhs.m_score;  //按照分数降序，排序
      } else {
        return lhs.m_id > rhs.m_id;  //相同分数，按照id排序
      }
    }
  }
};

int main() {
  std::set<Person, Cmp> mySet;
  Person s1(10, 100);
  Person s2(20, 200);
  Person s3(20, 300);
  Person s4(30, 200);
  Person s5(30, 300);
  mySet.insert(s1);  //插入s1
  mySet.insert(s2);  //插入s2
  mySet.insert(s3);  // s3和s2的id相同，不插入
  mySet.insert(s4);  //插入s4

  for (auto it : mySet) {
    std::cout << "id:" << it.m_id << ",score:" << it.m_score << std::endl;
  }
  std::cout << "end" << std::endl;
  return 0;
};
