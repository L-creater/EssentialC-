c++ 局部变量初始化，推荐用哪种写法？最不容易错？

auto 防止不初始化

#include<iostream>
#include <cstring>
#include <memory>
#include <vector>

using namespace std::literals;
struct C{
    int sex = 0;
    int age =18;
    std::string name = "hello"s;
};

struct lerp{
    float from =0;
    float to = 1;
    float fac = 0.5f;

    struct result{
        float value;
        float second_value;
    };

result  operator()()const{
    return{
    .value =  fac * (to - from) + from,
    .second_value = fac * (from - to) + to};
}
};


int main(int argc,char**argv){
    
    auto i = [&]{
        for(int i = 0;i < argc;++i){
            if(!strcmp(argv[i],"-h")){
                return i;
            }
        }
        return -1;
    }();

    int i1;  // 错误
    auto i2 = size_t(i1) ; //正确

    C c1; // 错误
    auto c2 = C{}; //正确
    auto p1 = new C;
    auto p2 = std::make_shared<C>(1,2); // 等价于 C(1,2); // 正确

    auto c3 = char(0); //解决方案
    auto s1 = std::string("hello"); // 解决方案
    auto s = "hello"s; // 更好的解决方案
    s= s + "world"s;
    std::cerr << s<<"\n";

    auto c = C{.age = 2};
    std::cerr << c.age<<"\n";
    std::cerr << c.name<<"\n"; 

    // 死循环
    // std::vector<int> a(0xffffff);
    // for(auto i =0; i <a.size();++i){
    //     std::cerr <<a[i];
    // }

    // todo static_cast 
    std::vector<int> a1(0xffffff);
    int *p; 
    // static_cast<size_t>(p)  // 会出错
    for(auto i = static_cast<size_t>(0); i <a1.size();++i){
        std::cerr <<a1[i];
    }
   

   //
    // 参数顺序问题解决方案，函数解决方法，
   auto x1 = lerp{.from = 0, .to =1, .fac = 0.25}();

    // 花里胡哨

    auto [x, y] = lerp{.from = 0, .to =1, .fac = 0.25}();
    std::cerr << x << "\n";
    std::cerr << y << "\n";
    return 0;
}
